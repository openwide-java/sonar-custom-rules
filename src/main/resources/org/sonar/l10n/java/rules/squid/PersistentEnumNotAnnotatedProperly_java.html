This rule detects usage enums not properly annoted

Noncompliant Code Example

@Entity
class MyCustomClass {

	@Enumerated // Noncompliant
	private MyEnum myField;
	
	@Enumerated(EnumType.ORDINAL)// Noncompliant
	private MyEnum myField2;
	
	private MyEnum myField3; // Noncompliant
	
}

Compliant Solution

@Entity
class MyCustomClass {

	@Enumerated(EnumType.STRING)
	private MyEnum myField1;
	
	@Enumerated(value = EnumType.STRING)
	private MyEnum myField2;
	
}
